use std;
use comm::oneshot;
use std::task::spawn;

fn main() {
    let mut sammiches = ~[];
    let (port, chan) = oneshot();
    do spawn {
        chan.send(make_sammich());
    }
    // parallel sammich production
    sammiches.push(make_sammich());
    sammiches.push(port.recv());
    do sammiches.len().times {
        sammiches.pop().eat();
    }
}
